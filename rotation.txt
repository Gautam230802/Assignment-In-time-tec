int strLen(char *stringFirst)
{
    int cnt =0;
    for(int i =0;stringFirst[i]!='\0';i++)
    {
        cnt++;
    }
    return cnt;
}
void concatFirst(char **stringFirst,char *stringSecond,int *capacity)
{
    int len =strLen(*stringFirst);
    int newLen =len;
    for(int i =0;stringSecond[i]!='\0';i++)
    {
        if((len+i) > *capacity)
        {
            *capacity =(*capacity)*2;
            *stringFirst =(char*)realloc(*stringFirst,(*capacity)*sizeof(char));
        }
        
        (*stringFirst)[newLen]=(*stringFirst)[i];
        newLen++;
    }
    (*stringFirst)[newLen+1]='\0';
    
}
