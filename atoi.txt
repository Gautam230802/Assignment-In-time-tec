#include<stdio.h>
#include<string.h>

int isDigit(char c)
{
    return (c>=48 && c<=57);
}
int stringToNum(char *num)
{   int ans =0;
    int sign =1;
    for(int i =0;num[i]!='\0';i++)
    { char c =num[i];
   
    if(num[i]=='-')
    {
        sign =-1;
    }
        else if(isDigit(c))
        {int temp =0;
            while(isDigit(num[i]))
            {
                temp =temp*10+(num[i]-'0');
                i++;
            }
            ans=temp*sign;
            sign =1;
        }
        else
        {
            return 0;
        }
    }
    return ans;
}
int main()
{
    char num[50];
    printf("enter the string: ");
    fgets(num,50,stdin);
    
    printf("%d",stringToNum(num));
    
}
